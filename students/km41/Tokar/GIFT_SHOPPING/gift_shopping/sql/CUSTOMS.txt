CREATE OR REPLACE TRIGGER giftorderitem 
  BEFORE INSERT ON GIFTS_GIFTITEMORDER 
    FOR EACH ROW
      BEGIN
       :NEW.ADD_TO_BUCKET_TIME := SYSDATE;
      END;
  /
  COMMIT;


CREATE OR REPLACE TRIGGER giftbucket_created_time 
BEFORE INSERT ON GIFTS_BUCKET 
FOR EACH ROW
BEGIN
 :NEW.CREATED_TIME := SYSDATE;
END;
/
COMMIT;
